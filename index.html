module U = struct
  let cat l = String.concat "" l
  let unwords l = String.concat " " l
  let unlines l = String.concat "\n" l
  let si = string_of_int
  let xyx x y = cat [x;y;x]
  let sq s = xyx "'" s
  let dq s = xyx "\"" s
end
open U

module F = struct
  let write f s =
    let c = open_out f in
    output_string c s;
    close_out c    
  let read f =  
    let lines = ref [] in
    let c = open_in f in
    try
      while true; do
        lines := input_line c :: !lines
      done; !lines
    with End_of_file -> close_in c; List.rev !lines
end
   
module H = struct          
  let el s ss = cat ["<";s;">";ss;""]
              
  let hn n s =
    let h = cat ["h";si n] in
    el h s

  let h1 s = hn 1 s
  let h2 s = hn 2 s
  let h3 s = hn 3 s
  let h4 s = hn 4 s
  let h5 s = hn 5 s
  let h6 s = hn 6 s

  let title s = el "title" s

  let kv k v = cat [k;"=";sq v]
  let kvn k n = cat [k;"=";sq @@ si n]
              
  let src s = kv "src" s
  let width n = kvn "width" n
  let height n = kvn "height" n
            
  let imgw u w = cat [""]
  let imgh u h = cat [""]

  let link u s = cat ["";s;""]
  let anchor s = cat [""]
               
  let li s = el "li" s
  let ul l = el "ul" @@ unlines @@ List.map li l
  let ol l = el "ol" @@ unlines @@ List.map li l

  let table l =
    let td x = el "td" x in
    let tr ll = el "tr" @@ unlines @@ List.map td ll in
    el "table" @@ unlines @@ List.map tr l
           
  let html h b =
    let head = el "head" @@ cat h in
    let body = el "body" @@ cat b in
    let s = cat [head;body] in
    el "html" s

  let page f h b = F.write f @@ html h b
end
